% Define a global flag, default is "both"
\newcommand{\languageflag}{both}

% Define the translation command for chapter titles
\newcommand{\translatechapter}[2]{
    \ifthenelse{\equal{\languageflag}{russian}}{
        \chapter{#1}
    }{
    \ifthenelse{\equal{\languageflag}{english}}{
        \chapter{#2}
    }{
        \chapter{#2 \hfill #1}
    }
    }
}

% Define the translation command that uses the global flag
\newcommand{\translate}[2]{
    \ifthenelse{\equal{\languageflag}{english}}{
        \begin{otherlanguage*}{english}
            #2
        \end{otherlanguage*}
    }{
    \ifthenelse{\equal{\languageflag}{russian}}{
        \begin{otherlanguage*}{russian}
            #1
        \end{otherlanguage*}
    }{
        \begin{minipage}[t]{0.45\linewidth}
            \begin{otherlanguage*}{english}
                #2
            \end{otherlanguage*}
        \end{minipage}%
        \hfill
        \begin{minipage}[t]{0.45\linewidth}
            \begin{otherlanguage*}{russian}
                #1
            \end{otherlanguage*}
        \end{minipage}
    }
    }
}
